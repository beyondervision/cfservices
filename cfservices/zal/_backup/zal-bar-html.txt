<script>
  /* ===============================
     Z.A.L. Context Sync (read-only)
     =============================== */

  document.addEventListener("DOMContentLoaded", () => {
    const input = document.getElementById("zal-input");
    const levelLabel = document.querySelector(".zal-level strong");

    if (window.CFS && CFS.aiGuidance) {
      const level = CFS.aiGuidance.level || "none";
      levelLabel.textContent = level;

      if (level !== "none") {
        input.disabled = false;
        input.placeholder =
          "Z.A.L. luistert mee (contextueel · nog geen acties)…";
      }
    }
  });
</script>
